\en{
  \cvitem{\textbf{FTP Client and Server}}{\textit{Computer Networks Course Project (A.Y. 2022-2023)} \newline
  Developed a multi-threaded \textbf{FTP-compliant} client-server application in \textbf{C} for file transfer, fully interoperable with \textbf{FileZilla}.
  \begin{itemize}
    \item Architecture: Multi-threaded server based on \textbf{Socket Programming} for concurrent session management.
    \item Data Modes: Native support for both Active (PORT) and Passive (PASV) data transfer modes.
    \item Protocol Management: Implemented essential \textit{FTP commands} (e.g., USER, PASS, LIST,RETR, STOR, CWD...)
    \item \textbf{Key skills:} \textbf{C}, \textbf{socket programming}, \textbf{multithreading}, \textbf{network protocols}, \textbf{FTP}, \textbf{Git}.
  \end{itemize}}
}
